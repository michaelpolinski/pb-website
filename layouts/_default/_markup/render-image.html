{{- /* AVIF/WebP logic from https://pawelgrzybek.com/webp-and-avif-images-on-a-hugo-website/ */ -}} {{- $width := 0 }} {{- $height := 0 }} {{- $invertDark := 0 }} {{- $u := urls.Parse .Destination }} {{- $u_path := $u.Path }} {{- with $u.Query }} {{- with .Get "height" }} {{- $height = int . }} {{- end }} {{- with .Get "width" }} {{- $width = int . }} {{- end }} {{- with .Get "invertDark" }} {{- $invertDark = int . }} {{- end }} {{ end }} {{ $isJPG := eq (path.Ext $u_path) ".jpg" }} {{ $isPNG := eq (path.Ext $u_path) ".png" }} {{ if ($isJPG) -}} {{ $avifPath := replace $u_path ".jpg" ".avif" }} {{ $avifPathStatic := printf "static/%s" $avifPath }} {{ if (fileExists $avifPathStatic) -}} {{- end }} {{ $webpPath := replace $u_path ".jpg" ".webp" }} {{ $webpPathStatic := printf "static/%s" $webpPath }} {{ if (fileExists $webpPathStatic) -}} {{- end }} {{- end }} {{ if ($isPNG) -}} {{ $avifPath := replace $u_path ".png" ".avif" }} {{ $avifPathStatic := printf "static/%s" $avifPath }} {{ if (fileExists $avifPathStatic) -}} {{- end }} {{ $webpPath := replace $u_path ".png" ".webp" }} {{ $webpPathStatic := printf "static/%s" $webpPath }} {{ if (fileExists $webpPathStatic) -}} {{- end }} {{- end }} {{- if and $width $height }} {{- else if and $width (not $height) }} {{- else if and $height (not $width) }} {{- else }} {{- end }}
